local dragging, dragInput, dragStart, startPos
Main.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true dragStart = input.Position startPos = Main.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
Main.InputChanged:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end end)
UIS.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        Main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

local function GetMyPlot()
    local Plots = workspace:FindFirstChild("Plots")
    if not Plots then return nil end
    for _, p in pairs(Plots:GetChildren()) do
        if p:GetAttribute("OwnerUserID") == LP.UserId then return p end
    end
    return nil
end

local FlySpeed = 50
local Flying = false
local FlyConnection = nil

local function ToggleFly()
    if not Toggles["Fly"] then
        if FlyConnection then FlyConnection:Disconnect() end
        Flying = false
        return
    end
    
    Flying = true
    local Character = LP.Character
    local Root = Character:FindFirstChild("HumanoidRootPart")
    
    if not Root then return end
    
    local BodyVelocity = Instance.new("BodyVelocity")
    BodyVelocity.Velocity = Vector3.new(0, 0, 0)
    BodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    BodyVelocity.Parent = Root
    
    local Camera = workspace.CurrentCamera
    
    if FlyConnection then FlyConnection:Disconnect() end
    FlyConnection = game:GetService("RunService").RenderStepped:Connect(function()
        if not Flying or not Root or not Root.Parent then
            BodyVelocity:Destroy()
            if FlyConnection then FlyConnection:Disconnect() end
            return
        end
        
        local moveDirection = Vector3.new(0, 0, 0)
        if UIS:IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + Camera.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - Camera.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection - Camera.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + Camera.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.Space) then moveDirection = moveDirection + Vector3.new(0, 1, 0) end
        if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then moveDirection = moveDirection - Vector3.new(0, 1, 0) end
        
        if moveDirection.Magnitude > 0 then
            moveDirection = moveDirection.Unit
        end
        
        BodyVelocity.Velocity = moveDirection * FlySpeed
    end)
end

local NoclipConnection = nil
local function ToggleNoclip()
    if not Toggles["Noclip"] then
        if NoclipConnection then NoclipConnection:Disconnect() end
        return
    end
    
    local Character = LP.Character
    if not Character then return end
    
    if NoclipConnection then NoclipConnection:Disconnect() end
    NoclipConnection = game:GetService("RunService").Stepped:Connect(function()
        if not Toggles["Noclip"] or not LP.Character then
            if NoclipConnection then NoclipConnection:Disconnect() end
            return
        end
        for _, part in pairs(LP.Character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end)
end

local SpeedConnection = nil
local function ToggleSpeed()
    if not Toggles["Speed"] then
        if SpeedConnection then SpeedConnection:Disconnect() end
        local Character = LP.Character
        if Character then
            local Humanoid = Character:FindFirstChild("Humanoid")
            if Humanoid then Humanoid.WalkSpeed = 16 end
        end
        return
    end
    
    local Character = LP.Character
    if not Character then return end
    local Humanoid = Character:FindFirstChild("Humanoid")
    if Humanoid then Humanoid.WalkSpeed = Settings["Speed"] end
    
    if SpeedConnection then SpeedConnection:Disconnect() end
    SpeedConnection = LP.CharacterAdded:Connect(function(NewCharacter)
        task.wait(0.1)
        local NewHumanoid = NewCharacter:FindFirstChild("Humanoid")
        if NewHumanoid then NewHumanoid.WalkSpeed = Settings["Speed"] end
    end)
end

local AntiAFKConnection = nil
local function ToggleAntiAFK()
    if not Toggles["Anti AFK"] then
        if AntiAFKConnection then AntiAFKConnection:Disconnect() end
        return
    end
    
    if AntiAFKConnection then AntiAFKConnection:Disconnect() end
    AntiAFKConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if Toggles["Anti AFK"] then
            local UserInputService = game:GetService("UserInputService")
            local Players = game:GetService("Players")
            local LocalPlayer = Players.LocalPlayer
            
            if not LocalPlayer.Character then return end
            
            local RandomKeypress = math.random(1, 6)
            local Keys = {Enum.KeyCode.W, Enum.KeyCode.A, Enum.KeyCode.S, Enum.KeyCode.D, Enum.KeyCode.Space}
            
            UserInputService:SendKeyEvent(true, Keys[RandomKeypress], false)
            task.wait(0.1)
            UserInputService:SendKeyEvent(false, Keys[RandomKeypress], false)
        end
    end)
end

task.spawn(function()
    local Knit = require(RS:WaitForChild("Packages"):WaitForChild("Knit"))
    local PlotService = Knit.GetService("PlotService")
    local StatService = Knit.GetService("StatUpgradeService")
    local Calculations = require(RS:WaitForChild("Modules"):WaitForChild("Calculations"))

    while task.wait(0.1) do
        local CashValue = LP:WaitForChild("leaderstats"):WaitForChild("Cash")
        local CurrentCash = CashValue:GetAttribute("CashAmount") or CashValue.Value or 0
        local Character = LP.Character
        local Root = Character and Character:FindFirstChild("HumanoidRootPart")
        local StatsFolder = LP:WaitForChild("updateStatsFolder")

        if Toggles["Auto Rebirth"] then
            pcall(function()
                local currentRebirths = StatsFolder:WaitForChild("Rebirths").Value
                local currentReach = StatsFolder:WaitForChild("Reach_Distance").Value
                local reachNeeded = currentRebirths * 30 + 80
                if currentReach >= reachNeeded and currentRebirths < 50 then
                    StatService:Rebirth()
                end
            end)
        end

        if Toggles["Auto Sell Brainrots [Click]"] then
            pcall(function()
                local Backpack = LP:FindFirstChild("Backpack")
                if Backpack then
                    for _, tool in pairs(Backpack:GetChildren()) do
                        local rarityLabel = tool:FindFirstChild("LevelBoard") and tool.LevelBoard:FindFirstChild("Frame") and tool.LevelBoard.Frame:FindFirstChild("Rarity")
                        if rarityLabel and rarityLabel:IsA("TextLabel") then
                            local rarityText = rarityLabel.Text:upper()
                            for selRarity, isSelected in pairs(SellSettings) do
                                if isSelected and rarityText:find(selRarity) then
                                    LP.Character:WaitForChild("Humanoid"):EquipTool(tool)
                                    RS.Remotes.SellBrainrot:FireServer("Single")
                                    task.wait(0.05)
                                    break
                                end
                            end
                        end
                    end
                end
            end)
        end

        if Toggles["Auto Collect"] then
            pcall(function()
                local myPlot = GetMyPlot()
                if myPlot and myPlot:FindFirstChild("Pods") then
                    for _, pod in pairs(myPlot.Pods:GetChildren()) do
                        local collectable = pod:GetAttribute("CollectAmount") or 0
                        if collectable > 0 then
                            PlotService:CollectCash(pod)
                            RS.Packages.Knit.Services.BrainrotService.RE.BrainrotCollected:FireServer(pod)
                            local tp = pod:FindFirstChild("Touch Part")
                            if tp then RS.Remotes.AnimateCollect:FireServer(collectable, tp) end
                        end
                    end
                end
            end)
        end

        local function GetMultiUpgradeCost(statName, currentLvl, amount)
            local totalCost = 0
            local tempLvl = currentLvl
            local step = (statName == "Reach_Distance") and 5 or 1
            for i = 1, amount do
                totalCost = totalCost + Calculations.getNextUpgradeCost(statName, tempLvl)
                tempLvl = tempLvl + step
            end
            return totalCost
        end

        local function HandleStatUpgrade(statName, toggleKey)
            if not Toggles[toggleKey] then return end
            pcall(function()
                local currentLvl = StatsFolder:WaitForChild(statName).Value
                if statName == "Power" or statName == "Reach_Distance" then
                    local amounts = {50, 25, 5, 1}
                    for _, amt in ipairs(amounts) do
                        local cost = GetMultiUpgradeCost(statName, currentLvl, amt)
                        if CurrentCash >= cost then StatService:Upgrade(statName, amt) break end
                    end
                else
                    local cost = Calculations.getNextUpgradeCost(statName, currentLvl)
                    if CurrentCash >= cost then StatService:Upgrade(statName, 1) end
                end
            end)
        end

        HandleStatUpgrade("Power", "Auto Power")
        HandleStatUpgrade("GrabAmount", "Auto Carry")
        HandleStatUpgrade("Reach_Distance", "Auto Reach")

        pcall(function()
            local ActiveBRs = workspace:FindFirstChild("ActiveBrainrots")
            if ActiveBRs and Root then
                for _, hitbox in pairs(ActiveBRs:GetChildren()) do
                    if hitbox.Name == "ServerHitbox" then
                        local model = hitbox:FindFirstChildOfClass("Model")
                        if model then
                            local rarityLabel = model:FindFirstChild("LevelBoard") and model.LevelBoard:FindFirstChild("Frame") and model.LevelBoard.Frame:FindFirstChild("Rarity")
                            if rarityLabel and rarityLabel:IsA("TextLabel") then
                                local txt = rarityLabel.Text:upper()
                                local shouldTeleport = false
                                if Toggles["Auto Mythic"] and txt:find("MYTHIC") then shouldTeleport = true
                                elseif Toggles["Auto Secret"] and txt:find("SECRET") then shouldTeleport = true
                                elseif Toggles["Auto Ancient"] and txt:find("ANCIENT") then shouldTeleport = true
                                elseif Toggles["Auto Divine"] and txt:find("DIVINE") then shouldTeleport = true end

                                if shouldTeleport then
                                    Root.CFrame = hitbox.CFrame * CFrame.new(0, 3, 0)
                                    task.wait(0.15)
                                    local attach = hitbox:FindFirstChild("Attachment")
                                    local prompt = attach and attach:FindFirstChild("ProximityPrompt")
                                    if prompt and prompt.ActionText == "PICK UP" then
                                        local oldHold = prompt.HoldDuration
                                        prompt.HoldDuration = 0
                                        prompt:InputHoldBegin()
                                        prompt:InputHoldEnd()
                                        prompt.HoldDuration = oldHold
                                        
                                        task.wait(0.2)
                                        local myPlot = GetMyPlot()
                                        if myPlot and myPlot:FindFirstChild("SpawnPart") then
                                            Root.CFrame = myPlot.SpawnPart.CFrame * CFrame.new(0, 4, 0)
                                        end
                                    end
                                    break
                                end
                            end
                        end
                    end
                end
            end
        end)

        if Toggles["Fly"] then ToggleFly() end
        if Toggles["Noclip"] then ToggleNoclip() end
        if Toggles["Speed"] then ToggleSpeed() end
        if Toggles["Anti AFK"] then ToggleAntiAFK() end
        if Toggles["Infinite Jump"] then ToggleInfiniteJump() end
        if Toggles["Shift Lock"] then ToggleShiftLock() end
        if Toggles["Anti Cheat"] then ToggleAntiCheat() end
        if Toggles["Auto Bat"] then ToggleAutoBat() end
        if Toggles["Auto Rejoin"] then ToggleAutoRejoin() end
    end
end)
